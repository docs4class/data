4.4 Top contenders… 

starwars <- starwars %>%
  group_by(homeworld) %>% 
  mutate(avg_bmi_by_hw = mean(BMI)) %>% 
  ungroup()


above_avg_BMI <- starwars %>%
  filter(BMI > starwars$avg_bmi_by_hw) %>%
  group_by(homeworld) %>%
  summarise(count = n()) %>%
  arrange(desc(count))
short_above <- head(above_avg_BMI, 7)
knitr::kable(short_above)
